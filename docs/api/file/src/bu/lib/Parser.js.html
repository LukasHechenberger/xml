    import { parser as createParser } from 'sax';
import Document from './Document';
import Element from './Element';
import { TextNode, CDataNode } from './Node';

const parsed = originalClass => class extends originalClass {

  static get name() {
    return `Parsed${super.name}`;
  }

  constructor(nodeOptions, parser) {
    super(nodeOptions);

    this._isSelfClosing = nodeOptions.isSelfClosing;
    this._position = parser.currentPosition;
  }

  toString() {
    return `${super.toString()}(${this._position.line}:${this._position.column})`;
  }
};

export const ParsedElement = parsed(Element);
// export class ParsedDocument = parsed(Document);
export const ParsedTextNode = parsed(TextNode);

export default class Parser {

  static get defaultOptions() {
    return {
      strict: false,
      lowercase: true,
    };
  }

  constructor(options = {}) {
    this._options = Object.assign({}, Parser.defaultOptions, options);
    this._parser = createParser(this._options);

    this._parser.ontext = this.onText.bind(this);
    this._parser.onopentag = this.onOpenTag.bind(this);
    this._parser.onclosetag = this.onCloseTag.bind(this);
    this._parser.oncdata = this.onCData.bind(this);

    this._document = new Document({});
    this._currentElement = this._document;
  }

  parse(string) {
    return new Promise((resolve, reject) => {
      this._parser.onerror = err => reject(err);

      this._parser.write(string);
      this._parser.close();

      // console.log(this._document);

      resolve({ document: this._document });
    });
  }

  get currentPosition() {
    return {
      line: this._parser.line,
      column: this._parser.column,
      position: this._parser.position,
    };
  }

  // SAX callbacks

  onText(text) {
    if (!this._currentElement) {
      console.log('Add to document textContent later', text);
    } else {
      this._currentElement.appendChild(new ParsedTextNode(text, this));
    }
  }

  onOpenTag(tag) {
    const elm = new ParsedElement(Object.assign({
      document: this._document,
      parent: this._currentElement,
    }, tag), this);

    if (this._currentElement === this._document) {
      this._document.documentElement = elm;
    }

    this._currentElement.appendChild(elm);

    this._currentElement = elm;
  }

  onCloseTag(tagName) {
    this._currentElement = this._currentElement.parentNode;
  }

  onCData(cdata) {
    this._currentElement.appendChild(new CDataNode(cdata));
  }

}
