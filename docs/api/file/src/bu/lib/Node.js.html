    import Builder from './Builder';

/**
 * @abstract
 */
export default class Node {

  constructor(options) {
    this._childNodes = [];
    this._nodeName = options.name;
    this._ownerDocument = options.document || null;
    this._parent = options.parent || null;
  }

  // Properties

  get childNodes() {
    return this._childNodes;
  }

  get firstChild() {
    return this._childNodes.length ? this._childNodes[0] : null;
  }

  get lastChild() {
    return this._childNodes.length ? this._childNodes[this._childNodes.length - 1] : null;
  }

  // FIXME: Implement nextSibling

  get nodeName() {
    return this._nodeName;
  }

  // FIXME: Insert nodeType

  get nodeValue() {
    return null;
  }

  set nodeValue(value) {
    return value;
  }

  get ownerDocument() {
    return this._ownerDocument;
  }

  get parentNode() {
    return this._parent || this._ownerDocument;
  }

  get parentElement() {
    const p = this._parent;
    return p && p.constructor._isElement ? p : null;
  }

  // FIXME: Insert previousSibling

  get textContent() {
    if (this.nodeValue) {
      return this.nodeValue;
    }

    return this._childNodes.map(n => n.textContent).join('');
  }

  set textContent(text) {
    this._childNodes = [new TextNode(text)];

    return text;
  }

  // Methods

  appendChild(childNode) {
    // FIXME: Detach child node
    // childNode._ownerDocument = this._ownerDocument;
    // childNode._parent = this._parent;

    this._childNodes.push(childNode);
  }

  // FIXME: Insert cloneNode()
  // FIXME: Insert hasChildNodes()
  // FIXME: Insert insertBefore()
  // FIXME: Insert isEqualNode()
  // FIXME: Insert isSameNode()
  // FIXME: Insert normalize()
  // FIXME: Insert removeChild()
  // FIXME: Insert replaceChild()

  // Non-Standard

  toString() {
    const c = this.childNodes.length ?
      this.childNodes.map(c => c.toString()).join('') :
      '';
    return `<${this.nodeName}>${c}`;
  }

  // Builder

  get [Builder.tagAttributes]() {
    return undefined;
  }

  [Builder.build](options) {
    let children = this._childNodes;

    children = children.filter(c => {
      return (c instanceof TextNode && c._text.trim() === '') ? false : true;
    });

    const indent = options.indent.repeat(options.level);
    const attributes = this[Builder.tagAttributes];
    const attributesString = attributes ? ` ${attributes}` : '';

    if (children.length) {
      const childOptions = Object.assign({}, options, {
        level: options.level + 1,
      });

      let childString = children.map(c => c[Builder.build](childOptions))
        .join(options.newline);

      if (children.length > 1 || !(children[0] instanceof TextNode)) {
        childString = `${options.newline}${childString}${options.newline}${indent}`;
      }

      return `${indent}<${this.nodeName}${attributesString}>${childString}</${this.nodeName}>`;
    }

    if (this._isSelfClosing) {
      return `${indent}<${this.nodeName}${attributesString} />`;
    }

    return `${indent}<${this.nodeName}${attributesString}></${this.nodeName}>`;
  }

}

export class TextNode extends Node {

  constructor(text) {
    super({
      name: '#text',
    });

    this._text = text;
  }

  get nodeValue() {
    return this._text;
  }

  set nodeValue(value) {
    return (this._text = value);
  }

  toString() {
    return this.textContent;
  }

  [Builder.build](options) {
    return this._text;
  }

}

export class CDataNode extends TextNode {

  [Builder.build](options) {
    return `<![CDATA[${this._text}]]>`;
  }

}
